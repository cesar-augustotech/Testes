import java.time.LocalDate;
import java.time.Period;
import java.time.format.DateTimeFormatter;
import java.util.ArrayList;
import java.util.List;

enum NivelHabilidade {
    INICIANTE("Iniciante"),
    INTERMEDIARIO("Intermediario"),
    AVANCADO("Avancado"),
    PROFISSIONAL("Profissional");

    private final String descricao;

    NivelHabilidade(String descricao) {
        this.descricao = descricao;
    }

    public String getDescricao() {
        return descricao;
    }
}

enum CategoriaInstrumento {
    CORDAS("Cordas"),
    SOPRO("Sopro"),
    TECLADO("Teclado"),
    PERCUSSAO("Percussao");

    private final String nome;

    CategoriaInstrumento(String nome) {
        this.nome = nome;
    }

    public String getNome() {
        return nome;
    }
}

abstract class Instrumento {

    private String modelo;
    private int anoFabricacao;
    private CategoriaInstrumento categoria;
    private LocalDate dataAquisicao;

    public Instrumento(String modelo, int anoFabricacao, CategoriaInstrumento categoria) {
        this.modelo = modelo;
        this.anoFabricacao = anoFabricacao;
        this.categoria = categoria;
        this.dataAquisicao = LocalDate.now();
    }

    public abstract void tocar();

    public void tocar(int intensidade) {
        System.out.println("[Intensidade " + intensidade + "/10] " + obterSomBase());
        if (intensidade >= 8) {
            System.out.println("Volume alto! Cuidado com as oreias.");
        }
    }

    public void tocar(String melodia) {
        System.out.println("Tocando melodia '" + melodia + "': " + obterSomBase());
    }

    protected abstract String obterSomBase();

    public int calcularIdade() {
        return LocalDate.now().getYear() - anoFabricacao;
    }

    public void exibirInfo() {
        DateTimeFormatter fmt = DateTimeFormatter.ofPattern("dd/MM/yyyy");
        System.out.println("  Instrumento : " + getClass().getSimpleName());
        System.out.println("  Modelo      : " + modelo);
        System.out.println("  Categoria   : " + categoria.getNome());
        System.out.println("  Fabricacao  : " + anoFabricacao + " (" + calcularIdade() + " anos)");
        System.out.println("  Adquirido em: " + dataAquisicao.format(fmt));
        System.out.println("  ---------------------------------");
    }

    public String getModelo() { return modelo; }
    public int getAnoFabricacao() { return anoFabricacao; }
    public CategoriaInstrumento getCategoria() { return categoria; }
    public LocalDate getDataAquisicao() { return dataAquisicao; }
}

class Piano extends Instrumento {

    private int numeraTeclas;

    public Piano(String modelo, int anoFabricacao, int numeraTeclas) {
        super(modelo, anoFabricacao, CategoriaInstrumento.TECLADO);
        this.numeraTeclas = numeraTeclas;
    }

    @Override
    public void tocar() {
        System.out.println("Piano '" + getModelo() + "': Plink... Plink... Plink...");
        System.out.println("(" + numeraTeclas + " teclas ressoando)");
    }

    @Override
    protected String obterSomBase() {
        return "Plink Plink Plink (" + numeraTeclas + " teclas)";
    }

    public int getNumeraTeclas() { return numeraTeclas; }
}

class Violao extends Instrumento {

    private int numeroNilon;
    private int numeroAco;

    public Violao(String modelo, int anoFabricacao, int numeroNilon, int numeroAco) {
        super(modelo, anoFabricacao, CategoriaInstrumento.CORDAS);
        this.numeroNilon = numeroNilon;
        this.numeroAco = numeroAco;
    }

    @Override
    public void tocar() {
        System.out.println("Violao '" + getModelo() + "': Tan... Ta... Tan Tan...");
        System.out.println("(" + (numeroNilon + numeroAco) + " cordas vibrando)");
    }

    @Override
    protected String obterSomBase() {
        return "Tan Ta Tan (" + numeroNilon + " nilon + " + numeroAco + " aco)";
    }
}

class Flauta extends Instrumento {

    private String material;

    public Flauta(String modelo, int anoFabricacao, String material) {
        super(modelo, anoFabricacao, CategoriaInstrumento.SOPRO);
        this.material = material;
    }

    @Override
    public void tocar() {
        System.out.println("Flauta '" + getModelo() + "': Fuuu... Fiii... Fuuu...");
        System.out.println("(flauta de " + material + ")");
    }

    @Override
    protected String obterSomBase() {
        return "Fuuu Fiii Fuuu (material: " + material + ")";
    }

    public String getMaterial() { return material; }
}

interface Musico {
    void praticar();
    String getNome();
    NivelHabilidade getNivel();
}

abstract class MusicoBase implements Musico {

    private String nome;
    private LocalDate dataNascimento;
    private NivelHabilidade nivel;
    private Instrumento instrumento;
    private LocalDate dataIngresso;

    public MusicoBase(String nome, LocalDate dataNascimento,
                      NivelHabilidade nivel, Instrumento instrumento) {
        this.nome = nome;
        this.dataNascimento = dataNascimento;
        this.nivel = nivel;
        this.instrumento = instrumento;
        this.dataIngresso = LocalDate.now();
    }

    public int calcularIdade() {
        return Period.between(dataNascimento, LocalDate.now()).getYears();
    }

    public String tempoNaEscola() {
        Period p = Period.between(dataIngresso, LocalDate.now());
        if (p.getYears() > 0)
            return p.getYears() + " ano(s) e " + p.getMonths() + " mes(es)";
        if (p.getMonths() > 0)
            return p.getMonths() + " mes(es)";
        return p.getDays() + " dia(s)";
    }

    public void exibirFicha() {
        DateTimeFormatter fmt = DateTimeFormatter.ofPattern("dd/MM/yyyy");
        System.out.println("  Nome       : " + nome);
        System.out.println("  Tipo       : " + getClass().getSimpleName());
        System.out.println("  Nascimento : " + dataNascimento.format(fmt) + " (" + calcularIdade() + " anos)");
        System.out.println("  Nivel      : " + nivel.getDescricao());
        System.out.println("  Instrumento: " + instrumento.getClass().getSimpleName() + " - " + instrumento.getModelo());
        System.out.println("  Na escola  : " + tempoNaEscola());
        System.out.println("  ---------------------------------");
    }

    public void trocarInstrumento(Instrumento novoInstrumento) {
        System.out.println(nome + " trocou de instrumento.");
        System.out.println("  Antes: " + instrumento.getClass().getSimpleName() + " (" + instrumento.getModelo() + ")");
        this.instrumento = novoInstrumento;
        System.out.println("  Agora: " + instrumento.getClass().getSimpleName() + " (" + instrumento.getModelo() + ")");
    }

    @Override public String getNome() { return nome; }
    @Override public NivelHabilidade getNivel() { return nivel; }
    public Instrumento getInstrumento() { return instrumento; }
    public LocalDate getDataNascimento() { return dataNascimento; }
}

class Pianista extends MusicoBase {

    public Pianista(String nome, LocalDate dataNascimento,
                    NivelHabilidade nivel, Piano piano) {
        super(nome, dataNascimento, nivel, piano);
    }

    @Override
    public void praticar() {
        System.out.println("Pianista " + getNome() + " pratica escalas e arpejos...");
        getInstrumento().tocar();
    }
}

class Guitarrista extends MusicoBase {

    public Guitarrista(String nome, LocalDate dataNascimento,
                       NivelHabilidade nivel, Violao violao) {
        super(nome, dataNascimento, nivel, violao);
    }

    @Override
    public void praticar() {
        System.out.println("Guitarrista " + getNome() + " pratica acordes e solos...");
        getInstrumento().tocar();
    }
}

class Flautista extends MusicoBase {

    public Flautista(String nome, LocalDate dataNascimento,
                     NivelHabilidade nivel, Flauta flauta) {
        super(nome, dataNascimento, nivel, flauta);
    }

    @Override
    public void praticar() {
        System.out.println("Flautista " + getNome() + " pratica respiracao e notas...");
        getInstrumento().tocar();
    }
}

public class EscolaDeMusica {

    private String nome;
    private LocalDate dataFundacao;
    private List<MusicoBase> alunos;
    private List<Instrumento> acervo;

    public EscolaDeMusica(String nome, LocalDate dataFundacao) {
        this.nome = nome;
        this.dataFundacao = dataFundacao;
        this.alunos = new ArrayList<>();
        this.acervo = new ArrayList<>();
    }

    public void adicionarAluno(MusicoBase musico) {
        alunos.add(musico);
        System.out.println("Aluno matriculado: " + musico.getNome());
    }

    public void adicionarAoAcervo(Instrumento instrumento) {
        acervo.add(instrumento);
    }

    public void realizarAula() {
        separador("AULA EM ANDAMENTO");
        for (MusicoBase aluno : alunos) {
            System.out.println("\n" + aluno.getNome() + " (" + aluno.getNivel().getDescricao() + "):");
            aluno.praticar();
        }
    }

    public void demonstrarSobrecarga() {
        separador("POLIMORFISMO DE SOBRECARGA");
        Instrumento inst = alunos.get(0).getInstrumento();
        System.out.println("\ntocar() padrao:");
        inst.tocar();
        System.out.println("\ntocar(int intensidade):");
        inst.tocar(9);
        System.out.println("\ntocar(String melodia):");
        inst.tocar("Fur Elise");
    }

    public void demonstrarTrocaInstrumentos() {
        separador("TROCA DE INSTRUMENTOS");
        if (alunos.size() >= 2) {
            MusicoBase a1 = alunos.get(0);
            MusicoBase a2 = alunos.get(1);
            Instrumento instA1 = a1.getInstrumento();
            Instrumento instA2 = a2.getInstrumento();
            System.out.println();
            a1.trocarInstrumento(instA2);
            a2.trocarInstrumento(instA1);
            System.out.println("\nApos a troca:");
            a1.praticar();
            a2.praticar();
        }
    }

    public void exibirFichasAlunos() {
        separador("FICHAS DOS ALUNOS");
        for (MusicoBase aluno : alunos) {
            System.out.println();
            aluno.exibirFicha();
        }
    }

    public void exibirAcervo() {
        separador("ACERVO DE INSTRUMENTOS");
        for (Instrumento inst : acervo) {
            System.out.println();
            inst.exibirInfo();
        }
    }

    public void exibirCabecalho() {
        int anos = Period.between(dataFundacao, LocalDate.now()).getYears();
        System.out.println("==============================================");
        System.out.println("  " + nome);
        System.out.println("  Fundada em " + dataFundacao.getYear() + " - " + anos + " anos");
        System.out.println("  Alunos: " + alunos.size() + "  |  Acervo: " + acervo.size() + " instrumentos");
        System.out.println("==============================================");
    }

    private void separador(String titulo) {
        System.out.println("\n------------------------------------------");
        System.out.println("  " + titulo);
        System.out.println("------------------------------------------");
    }

    public static void main(String[] args) {

        Piano pianoSteinway       = new Piano("Steinway D-274", 2018, 88);
        Violao violaoYamaha       = new Violao("Yamaha FG800", 2020, 3, 3);
        Flauta flautaGemeinhardt  = new Flauta("Gemeinhardt 2SP", 2019, "metal");
        Flauta flautaExtra        = new Flauta("Jupiter JFL-700", 2022, "prata");

        Pianista ana = new Pianista(
            "Ana Souza",
            LocalDate.of(2005, 3, 15),
            NivelHabilidade.AVANCADO,
            pianoSteinway
        );

        Guitarrista bruno = new Guitarrista(
            "Bruno Lima",
            LocalDate.of(2008, 7, 22),
            NivelHabilidade.INTERMEDIARIO,
            violaoYamaha
        );

        Flautista carla = new Flautista(
            "Carla Neves",
            LocalDate.of(2003, 11, 5),
            NivelHabilidade.PROFISSIONAL,
            flautaGemeinhardt
        );

        EscolaDeMusica escola = new EscolaDeMusica(
            "Escola Harmonia Musical",
            LocalDate.of(2010, 8, 1)
        );

        System.out.println("Matriculando alunos...");
        escola.adicionarAluno(ana);
        escola.adicionarAluno(bruno);
        escola.adicionarAluno(carla);

        escola.adicionarAoAcervo(pianoSteinway);
        escola.adicionarAoAcervo(violaoYamaha);
        escola.adicionarAoAcervo(flautaGemeinhardt);
        escola.adicionarAoAcervo(flautaExtra);

        escola.exibirCabecalho();
        escola.exibirAcervo();
        escola.exibirFichasAlunos();
        escola.realizarAula();
        escola.demonstrarSobrecarga();
        escola.demonstrarTrocaInstrumentos();

        System.out.println("\nFim da demonstracao.\n");
    }
}
